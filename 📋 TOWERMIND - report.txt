DATABASE SCHEMA
Table: resources
sql
CREATE TABLE resources (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    floor VARCHAR(20) NOT NULL,
    room VARCHAR(50),
    resource_type VARCHAR(20) NOT NULL,  -- 'electricity', 'water'
    value DECIMAL(10, 2) NOT NULL,
    cost DECIMAL(10, 2),
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for performance
CREATE INDEX idx_resources_floor ON resources(floor);
CREATE INDEX idx_resources_timestamp ON resources(timestamp);
CREATE INDEX idx_resources_type ON resources(resource_type);
Table: bookings
sql
CREATE TABLE bookings (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    floor VARCHAR(20) NOT NULL,
    room_name VARCHAR(50) NOT NULL,
    booked_by VARCHAR(50) NOT NULL,
    start_time DATETIME NOT NULL,
    end_time DATETIME NOT NULL,
    status VARCHAR(20) DEFAULT 'booked',  -- 'booked', 'cancelled', 'completed'
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_bookings_room ON bookings(room_name);
CREATE INDEX idx_bookings_time ON bookings(start_time, end_time);
Table: anomalies
sql
CREATE TABLE anomalies (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    floor VARCHAR(20) NOT NULL,
    room VARCHAR(50),
    anomaly_type VARCHAR(30) NOT NULL,  -- 'energy_spike', 'water_leak', 'overnight_usage', 'equipment_failure'
    message TEXT NOT NULL,
    severity VARCHAR(20) NOT NULL,  -- 'critical', 'high', 'medium', 'low'
    status VARCHAR(20) DEFAULT 'pending',  -- 'pending', 'acknowledged', 'resolved'
    diagnostics TEXT,  -- JSON string of possible causes
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    resolved_at DATETIME
);

CREATE INDEX idx_anomalies_severity ON anomalies(severity);
CREATE INDEX idx_anomalies_status ON anomalies(status);
Table: recommendations
sql
CREATE TABLE recommendations (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title VARCHAR(100) NOT NULL,
    description TEXT NOT NULL,
    category VARCHAR(20) NOT NULL,  -- 'electricity', 'water', 'space', 'manpower', 'materials'
    estimated_savings DECIMAL(10, 2),
    confidence_score DECIMAL(5, 2),  -- 0-100
    carbon_impact DECIMAL(10, 2),  -- kg CO₂
    carbon_equivalent VARCHAR(100),  -- e.g., "22 mature trees"
    implemented BOOLEAN DEFAULT FALSE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    implemented_at DATETIME
);

CREATE INDEX idx_recommendations_category ON recommendations(category);
CREATE INDEX idx_recommendations_implemented ON recommendations(implemented);
Table: kpi_history
sql
CREATE TABLE kpi_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    month DATE NOT NULL,
    efficiency_score INTEGER NOT NULL,  -- 0-100
    energy_score INTEGER,  -- 0-100
    carbon_score INTEGER,  -- 0-100
    resource_score INTEGER,  -- 0-100
    carbon_footprint DECIMAL(10, 2),  -- tons CO₂
    total_savings DECIMAL(10, 2),  -- RM
    cost_reduction DECIMAL(5, 2),  -- percentage
    UNIQUE(month)
);

CREATE INDEX idx_kpi_month ON kpi_history(month);
Table: simulation_logs
sql
CREATE TABLE simulation_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_query TEXT NOT NULL,
    scenario_a JSON,
    scenario_b JSON,
    scenario_c JSON,
    recommended_scenario VARCHAR(10),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
API ENDPOINTS
Resources API
text
GET /api/resources
  Query params: floor, resource_type, start_date, end_date
  Response: Array of resource objects

GET /api/resources/summary
  Response: Current totals by type

GET /api/resources/trend
  Query params: days (default 7), resource_type
  Response: Aggregated trend data
Bookings API
text
GET /api/bookings
  Query params: start_date, end_date
  Response: Array of bookings

POST /api/bookings
  Body: { floor, room_name, booked_by, start_time, end_time }
  Response: Created booking object

PUT /api/bookings/{id}/status
  Body: { status: 'cancelled' or 'completed' }
  Response: Updated booking
Anomalies API
text
GET /api/anomalies
  Query params: severity, status, limit
  Response: Array of anomalies

PUT /api/anomalies/{id}/status
  Body: { status: 'acknowledged' or 'resolved' }
  Response: Updated anomaly

GET /api/anomalies/detect
  Response: Runs detection algorithm and returns new anomalies
Recommendations API
text
GET /api/recommendations
  Query params: category, implemented
  Response: Array of recommendations

PUT /api/recommendations/{id}/implement
  Response: Updates implemented status

POST /api/recommendations/generate
  Response: Runs engine and creates new recommendations
KPI API
text
GET /api/kpi/current
  Response: Current month KPI

GET /api/kpi/history
  Query params: months (default 6)
  Response: Array of monthly KPI data

GET /api/kpi/leaderboard
  Response: Floors ranked by efficiency
Forecast API
text
GET /api/forecast/next-month
  Response: { projected_cost, growth_percentage, drivers, confidence }

GET /api/forecast/trend
  Query params: days (default 30)
  Response: Daily forecast data
Simulation API
text
POST /api/simulation/generate
  Body: { query: "How to reduce energy by 15%" }
  Response: { scenario_a, scenario_b, scenario_c, recommended, analysis }

POST /api/simulation/fallback
  Body: { query: "..." }
  Response: Rule-based scenarios (fallback)

- Python 3.8+
- Node.js 16+ (for frontend development)
- Gemini API Key (Google AI Studio)

## Installation

### 1. Clone Repository
```bash
git clone https://github.com/yourusername/TowerMind.git
cd TowerMind
2. Backend Setup
bash
cd backend
python -m venv venv
source venv/bin/activate  # On Windows: venv\Scripts\activate
pip install -r requirements.txt
3. Configure Environment
bash
cp .env.example .env
# Edit .env and add your GEMINI_API_KEY
4. Initialize Database
bash
python data/init_db.py
5. Start Backend
bash
python app.py
# Server runs at http://localhost:5000
6. Frontend Setup
bash
cd ../frontend
# No build step - just open index.html in browser
# Or use Live Server extension in VS Code
7. Access Application
Open frontend/index.html in your browser

API will connect to http://localhost:5000

Default Demo Data
8 floors, 20 rooms

30 days of resource data

Pre-populated anomalies and recommendations

API Documentation
API endpoints available at http://localhost:5000/api/
